def solution(n, arr1, arr2):
    answer = []
    for v1, v2 in zip(arr1, arr2):
        tmp = bin(v1 | v2)[2:].replace('1', '#').replace('0',' ')
        answer.append(' ' * (n - len(tmp)) + tmp) # == answer.append(tmp.rjust(n, ' '))
    return answer
