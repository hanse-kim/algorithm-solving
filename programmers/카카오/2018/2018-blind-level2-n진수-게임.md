base = '0123456789ABCDEF'

def baseConversion(num, n):
    if num == 0:
        return ['0']

    result = []
    while (num > 0):
        num, remainder = divmod(num, n)
        result.append(base[remainder])

    return result[::-1]

def solution(n, t, m, p):
    totalNums = []
    targetCount = p + (t - 1) * m
    i = 0
    while (len(totalNums) < targetCount):
        totalNums.extend(baseConversion(i, n))
        i += 1
    
    result = [totalNums[i] for i in range(p - 1, len(totalNums), m)]
    return ''.join(result[:t])
