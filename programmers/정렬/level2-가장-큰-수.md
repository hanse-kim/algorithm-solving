import functools

def compare(x:str, y:str) -> bool:
    return 1 if x + y < y + x else -1

def solution(numbers):
    if sum(numbers) == 0:
        return "0"
    
    return "".join(sorted([str(n) for n in numbers], key=functools.cmp_to_key(compare)))
