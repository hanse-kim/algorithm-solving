def solution(prices):
    result = [ i for i in range(len(prices) - 1, -1, -1) ]
    stack = []
    for i, price in enumerate(prices):
        while stack and prices[stack[-1]] > price:
            prev = stack.pop()
            result[prev] = i - prev
            
        stack.append(i)
        
    return result
